127bool bike_process::process()
128{
129 int bike_h, times_h;
130 SDL_QueryTexture(bike, NULL, NULL, NULL, &bike_h);
131 SDL_QueryTexture(times, NULL, NULL, NULL, &times_h);
132 while( SDL_PollEvent(&event) )
133 {
134 if(event.type==SDL_QUIT)
135 return false;
136 }
137
138 if(keystate[SDL_SCANCODE_ESCAPE] || game_on == 0 || time_gap >= 90) //return to menu
139 return false;
140
141 if(keystate[SDL_SCANCODE_SPACE] or keystate[SDL_SCANCODE_UP])//If space or arrow up is pressed...
142 {
143 if(jump_state == 0) //if we are not in jump already
144 jump_state = 36; //set the jump or SPACE state in 20*2 steps...
145 }
146 if( (bike_y <(bike_stand-bike_h)) or jump_state >= 16 ) //Condition true untill the bike goes upward and downward to surface below it..
147 {
148 int jump_height;
149 jump_height = 19 - jump_state; //using the formula n/2*(n+1)...
150 bike_y += jump_height;
151 jump_state--;
152 if(jump_state == 0)
153 jump_state--;
154 }
155 else
156 {
157 bike_y = bike_stand-bike_h;
158 jump_state = 0;
159 }
160
161 if(keystate[SDL_SCANCODE_RIGHT])
162 if(speed<=20) //Speed only increase to 20 (in fact, to 10 as while changing coordinates speed is divided by 2..
163 speed++;
164 if(keystate[SDL_SCANCODE_LEFT])
165 if(speed>0) //speed cannot be less than 0..
166 speed--;
167
168 //cout most info about texture state
169 if (keystate[SDL_SCANCODE_A])
170 {
171 cerr << "speed: " << speed << std::endl;
172 cerr << "bike_x: " << bike_x << std::endl;
173 cerr << "bike_y: " << bike_y << std::endl;
174 cerr << "bike_h: " << bike_h << std::endl;
175 cerr << "jump_state: " << jump_state << std::endl;
176 cerr << "timegap: " << time_gap << std::endl << std::endl;
177 for (int i = 0; i < SMAX_WALL; i++)
178 {
179 cerr << "w1[" << i << "]: \n";
180 cerr << "state: " << w1[i].get_state() << "\n";
181 cerr << "x: " << w1[i].get_cord_x() << "\n";
182 cerr << "y: " << w1[i].get_cord_y() << "\n";
183 cerr << "w: " << w1[i].get_cord_w() << "\n";
184 cerr << "h: " << w1[i].get_cord_h() << "\n\n";
185 }
186 for (int i = 0; i < BMAX_WALL; i++)
187 {
188 cerr << "w1[" << i << "]: \n";
189 cerr << "state: " << w2[i].get_state() << "\n";
190 cerr << "x: " << w2[i].get_cord_x() << "\n";
191 cerr << "y: " << w2[i].get_cord_y() << "\n";
192 cerr << "w: " << w2[i].get_cord_w() << "\n";
193 cerr << "h: " << w2[i].get_cord_h() << "\n\n";
194 }
195 for (int i = 0; i < 2; i++)
196 {
197 cerr << "bg[" << i << "]: \n";
198 cerr << "x: " << bg[i].get_cord_x() << "\n";
199 cerr << "y: " << bg[i].get_cord_y() << "\n";
200 cerr << "w: " << bg[i].get_cord_w() << "\n";
201 cerr << "h: " << bg[i].get_cord_h() << "\n\n";
202 }
203 }
204
205 if(bike_x < FINISH_BIKE_X) //at the beginning bike's texture moves around the screen
206 bike_x+=speed;
207 else//and afterwards its texture remains static but everything moves instead
208 change_coordinate(-speed/2);
209 change_coordinate(0);//background moves regardless of speed
210 SDL_RenderClear(renderer);
211 load_layer();
212 if(finished())
213 {
214 finish();
215 return false;
216 }
217
218 std::stringstream time;
219 //Convert the timer's time to a string
220 time_gap =(SDL_GetTicks() - start_time)/1000;
221 int mm=time_gap/60;
222 int ss=time_gap%60;
223 if(ss < 10)
224 time << "Timer: " << mm << ":0" << ss;
225 else
226 time << "Timer: " << mm << ":" << ss;
227 times = SDL_CreateTextureFromSurface(renderer, TTF_RenderText_Solid(font,time.str().c_str(),textColor));
228 apply_surface(600, 50, times, renderer);
229 apply_surface(500, 55+times_h, time_limit, renderer);
230 SDL_Delay(40);
